table1 = product

columns = productId, productName

table2 = productOrder
columns = orderId, productId, insertedTime

table3 = orderStage
columns = orderStageId, orderId, stage

*each order has different stages

Show all orders on stage2 and product name

SELECT o.orderId, productName
FROM product p 
INNER JOIN productOrder o
ON p.productId = o.productId
INNER JOIN orderStage s
ON s.orderId = o.orderId
WHERE stage = '2'

SELECT MAX(stage), o.orderId, productName
FROM product p 
INNER JOIN productOrder o
ON p.productId = o.productId
INNER JOIN orderStage s
ON s.orderId = o.orderId
GROUP BY o.orderId, productName
